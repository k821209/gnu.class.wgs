권장 필터링 기준

SNP 필터링

  • QUAL: ≥ 30 (99.9% 신뢰도)
  • DP: 10-100 (적절한 깊이)
  • QD: ≥ 2.0 (깊이당 품질)
  • FS: ≤ 60.0 (스트랜드 편향)
  • MQ: ≥ 40.0 (매핑 품질)

INDEL 필터링

  • QUAL: ≥ 20 (99% 신뢰도)
  • DP: 8-80 (INDEL은 더 관대)
  • QD: ≥ 1.0 (INDEL 특성 고려)
  • FS: ≤ 200.0 (INDEL 허용치)
  • ReadPosRankSum: ≥ -8.0

# !pip install koreanize-matplotlib 터미널에 입력하여 설치 # 설치 후 import로 불러오기 import pandas as pd import matplotlib.pyplot as plt import seaborn as sns import numpy as np import subprocess from collections import Counter import gzip import koreanize_matplotlib def comprehensive_variant_analysis(vcf_file, title="Variant Analysis"): """포괄적인 변이 분석을 수행합니다.""" print(f"🔬 {title}") print("=" * 60) # VCF 파싱 variants = [] if vcf_file.endswith('.gz'): file_opener = gzip.open mode = 'rt' else: file_opener = open mode = 'r' with file_opener(vcf_file, mode) as f: for line in f: if line.startswith('#'): continue fields = line.strip().split('\t') if len(fields) < 10: continue chrom = fields[0] pos = int(fields[1]) ref = fields[3] alt = fields[4] qual = float(fields[5]) if fields[5] != '.' else 0 info = fields[7] format_field = fields[8] sample = fields[9] # 변이 유형 결정 if len(ref) == 1 and len(alt) == 1: variant_type = 'SNP' transition = classify_transition(ref, alt) elif len(ref) > len(alt): variant_type = 'Deletion' transition = 'INDEL' elif len(ref) < len(alt): variant_type = 'Insertion' transition = 'INDEL' else: variant_type = 'Complex' transition = 'Complex' # 깊이 및 유전자형 정보 추출 dp = extract_dp_from_info(info) gt = extract_genotype(sample) variants.append({ 'chrom': chrom, 'pos': pos, 'ref': ref, 'alt': alt, 'qual': qual, 'type': variant_type, 'transition': transition, 'depth': dp, 'genotype': gt }) df = pd.DataFrame(variants) return df def classify_transition(ref, alt): """SNP의 transition/transversion 분류""" transitions = [('A', 'G'), ('G', 'A'), ('C', 'T'), ('T', 'C')] if (ref, alt) in transitions: return 'Transition' else: return 'Transversion' def extract_dp_from_info(info): """INFO 필드에서 DP 값 추출""" for item in info.split(';'): if item.startswith('DP='): return int(item.split('=')[1]) return 0 def extract_genotype(sample): """샘플 필드에서 유전자형 추출""" gt = sample.split(':')[0] if gt in ['0/0']: return 'Homozygous Reference' elif gt in ['0/1', '1/0']: return 'Heterozygous' elif gt in ['1/1']: return 'Homozygous Variant' else: return 'Other' def create_comprehensive_plots(df, title): """종합적인 시각화 생성""" fig, axes = plt.subplots(3, 2, figsize=(16, 18)) fig.suptitle(f'{title} - Comprehensive Analysis', fontsize=16, fontweight='bold') # 1. 변이 유형별 분포 type_counts = df['type'].value_counts() axes[0, 0].pie(type_counts.values, labels=type_counts.index, autopct='%1.1f%%', colors=['#3498db', '#e74c3c', '#f39c12', '#9b59b6']) axes[0, 0].set_title('변이 유형별 분포') # 2. Transition/Transversion 비율 snp_df = df[df['type'] == 'SNP'] if not snp_df.empty: ti_tv = snp_df['transition'].value_counts() axes[0, 1].bar(ti_tv.index, ti_tv.values, color=['#2ecc71', '#e67e22']) axes[0, 1].set_title(f'Ti/Tv 비율: {ti_tv["Transition"]/ti_tv["Transversion"]:.2f}') axes[0, 1].set_ylabel('SNP 개수') # 3. 염색체별 변이 밀도 chrom_counts = df['chrom'].value_counts().head(10) axes[1, 0].bar(range(len(chrom_counts)), chrom_counts.values) axes[1, 0].set_xticks(range(len(chrom_counts))) axes[1, 0].set_xticklabels(chrom_counts.index, rotation=45) axes[1, 0].set_title('염색체별 변이 개수 (상위 10개)') axes[1, 0].set_ylabel('변이 개수') # 4. 품질 점수 분포 quality_data = df[df['qual'] > 0]['qual'] axes[1, 1].hist(quality_data, bins=50, alpha=0.7, color='#3498db') axes[1, 1].axvline(x=30, color='red', linestyle='--', label='Q30 임계값') axes[1, 1].set_title('변이 품질 점수 분포') axes[1, 1].set_xlabel('QUAL Score') axes[1, 1].set_ylabel('빈도') axes[1, 1].legend() # 5. 깊이 분포 depth_data = df[df['depth'] > 0]['depth'] axes[2, 0].hist(depth_data, bins=50, alpha=0.7, color='#e74c3c') axes[2, 0].axvline(x=depth_data.mean(), color='green', linestyle='--', label=f'평균: {depth_data.mean():.1f}') axes[2, 0].set_title('변이 깊이 분포') axes[2, 0].set_xlabel('Read Depth') axes[2, 0].set_ylabel('빈도') axes[2, 0].legend() # 6. 유전자형 분포 gt_counts = df['genotype'].value_counts() axes[2, 1].bar(gt_counts.index, gt_counts.values, color=['#9b59b6', '#f39c12', '#2ecc71', '#95a5a6']) axes[2, 1].set_title('유전자형 분포') axes[2, 1].set_ylabel('변이 개수') axes[2, 1].tick_params(axis='x', rotation=45) plt.tight_layout() plt.savefig(f'{title.lower().replace(" ", "_")}_analysis.png', dpi=300, bbox_inches='tight') plt.show() return df def generate_summary_statistics(df): """요약 통계 생성""" print(f"\n📊 요약 통계:") print(f"총 변이 수: {len(df):,}") print(f"SNP: {len(df[df['type'] == 'SNP']):,}") print(f"Insertion: {len(df[df['type'] == 'Insertion']):,}") print(f"Deletion: {len(df[df['type'] == 'Deletion']):,}") if len(df[df['type'] == 'SNP']) > 0: snp_df = df[df['type'] == 'SNP'] ti_count = len(snp_df[snp_df['transition'] == 'Transition']) tv_count = len(snp_df[snp_df['transition'] == 'Transversion']) print(f"Ti/Tv 비율: {ti_count/tv_count:.2f}") print(f"평균 품질 점수: {df[df['qual'] > 0]['qual'].mean():.2f}") print(f"평균 깊이: {df[df['depth'] > 0]['depth'].mean():.1f}") # 유전자형 분포 gt_dist = df['genotype'].value_counts() total = len(df) print(f"\n🧬 유전자형 분포:") for gt, count in gt_dist.items(): print(f"{gt}: {count:,} ({count/total*100:.1f}%)") # 분석 실행 filtered_gatk_file = '/home/ubuntu/edu/soybean_data/variant_calling/soybean_gatk_filtered.vcf.gz' filtered_bcftools_file = '/home/ubuntu/edu/soybean_data/variant_calling/soybean_bcftools_filtered.vcf.gz' # GATK 필터링 결과 분석 gatk_df = comprehensive_variant_analysis(filtered_gatk_file, "GATK Filtered Results") gatk_stats = create_comprehensive_plots(gatk_df, "GATK Filtered") generate_summary_statistics(gatk_df) print("\n" + "="*80 + "\n") # bcftools 필터링 결과 분석 bcftools_df = comprehensive_variant_analysis(filtered_bcftools_file, "bcftools Filtered Results") bcftools_stats = create_comprehensive_plots(bcftools_df, "bcftools Filtered") generate_summary_statistics(bcftools_df)

생물학적 의미 해석

HIGH Impact 변이

  • 단백질 기능 손실: 조기 종료, 프레임시프트
  • 생물학적 중요성: 표현형 변화 가능성 높음
  • 식물 육종: 중요 형질 관련 후보
  • 우선순위: 최우선 검증 대상

MODERATE Impact 변이

  • 아미노산 변화: 단백질 기능 변화
  • 생물학적 중요성: 조건부 영향 가능
  • 식물 육종: 양적 형질 관련 후보
  • 우선순위: 2차 검증 대상
